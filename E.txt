package h;

public class E {

    E(int a,int b){
        if(a>b)
            System.out.println(a);
        else
            System.out.println(b);
    }

    E(double a,double b,double c){
        if(a==b&&b==c){
            System.out.println("相等");
        }else
            System.out.println("不相等");
    }

    public static void main(String[] args) {
        E e1 = new E(1,2);
        E e2 = new E(1.1,2.3,1.1);
    }
}
