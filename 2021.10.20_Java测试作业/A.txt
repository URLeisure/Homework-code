package h;

import java.util.Scanner;

public class A {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int n = input.nextInt();
        int a = input.nextInt();
        int sum = 0,temp = a;
        for(int i = 0;i<n;i++){
            sum += temp;
            //System.out.println(temp);
            temp = temp*10+a;
        }
        System.out.println(sum);
    }
}
